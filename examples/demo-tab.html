📏 tab-size 缩进宽度对比

tab-size: 2

function renderList(items) {
	if (!items.length) {
		return null;
	}

	for (let i = 0; i < items.length; i++) {
		const item = items[i];
		console.log(item.name, item.value);
	}

	return true;
}
